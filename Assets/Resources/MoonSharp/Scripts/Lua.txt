
local rotate = function(self, delta)
  self._game_object.transform.Rotate(0, 0, self._speed * delta);
end -- rotate

local translate = function(self, vector)
    self._game_object.transform.Translate(vector);
end

local rotateVector= function(self, vector)
    self._game_object.transform.Rotate(vector);
end


function make (game_object, speed)
  return 
  {
    _game_object = game_object;
    _speed = speed;

    rotate = rotate;
	rotateVector=rotateVector;
	translate=translate;
  }
end

--

return
{
  make = make;
}
